<div id="game" class="screen">
  <h2 id="className"></h2>
  <p id="stats"></p>
  <button onclick="attack()">Attaquer le boss</button>
  <p id="log"></p>
</div>

<script>
const classes = {
  furtif: { name:'Assassin furtif', atk:5, hp:20 },
  techno: { name:'Assassin techno', atk:4, hp:25 },
  plasma: { name:'Archer plasma', atk:6, hp:18 }
};
const boss = { name:'Golem de Fer', hp:50, atk:3 };
let player, bossHp;

function start(key) {
  player = { ...classes[key] };
  bossHp = boss.hp;
  document.getElementById('className').textContent = player.name;
  updateStats();
  document.getElementById('log').textContent = '';
  show('game');
}

function updateStats() {
  document.getElementById('stats').textContent =
    `PV : ${player.hp} | Attaque : ${player.atk} ‚Äî Boss PV : ${bossHp}`;
}

function attack() {
  bossHp -= player.atk;
  let msg = `Tu frappes le boss (-${player.atk} PV).\n`;
  if (bossHp > 0) {
    player.hp -= boss.atk;
    msg += `Le boss te riposte (-${boss.atk} PV).\n`;
  }
  if (bossHp <= 0) msg += 'üéâ Boss vaincu !';
  else if (player.hp <= 0) msg += 'üíÄ Tu as √©t√© vaincu...';
  document.getElementById('log').textContent = msg;
  updateStats();
}

function show(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById(id).classList.add('active');
}
</script>
